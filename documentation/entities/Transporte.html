import {
  Entity,
  Enum,
  ManyToOne,
  type Opt,
  PrimaryKey,
  Property,
  Unique,
} from '@mikro-orm/core';
import { Center } from 'src/center/domain/entities/Center.entity';
import { User } from 'src/user/domain/entities/User.entity';

@Entity({ tableName: 'Transporte' })
export class Transporte {
  @PrimaryKey()
  id!: number;

  @Unique({
    name: 'Transporte_numeroTransporte_key',
    expression:
      'CREATE UNIQUE INDEX "Transporte_numeroTransporte_key" ON public."Transporte" USING btree ("numeroTransporte")',
  })
  @Property({ fieldName: 'numeroTransporte', type: 'text' })
  numeroTransporte!: string;

  @Enum({ items: () => TransporteStatus })
  status: TransporteStatus & Opt = TransporteStatus.AGUARDANDO_SEPARACAO;

  @Property({ fieldName: 'nomeRota', type: 'text' })
  nomeRota!: string;

  @Property({ fieldName: 'nomeTransportadora', type: 'text' })
  nomeTransportadora!: string;

  @Property({ type: 'text' })
  placa!: string;

  @Property({
    fieldName: 'criadoEm',
    type: 'datetime',
    columnType: 'timestamp(3)',
    defaultRaw: `CURRENT_TIMESTAMP`,
  })
  criadoEm!: Date & Opt;

  @Property({ fieldName: 'atualizadoEm', columnType: 'timestamp(3)' })
  atualizadoEm!: Date;

  @ManyToOne({
    entity: () => User,
    fieldName: 'cadastradoPorId',
    updateRule: 'cascade',
  })
  cadastradoPorId!: User;

  @Property({ fieldName: 'dataExpedicao', columnType: 'timestamp(6)' })
  dataExpedicao!: Date;

  @ManyToOne({
    entity: () => Center,
    fieldName: 'centerId',
    updateRule: 'cascade',
  })
  centerId!: Center;

  @Property({ type: 'text', nullable: true })
  obs?: string;

  @Property({ type: 'integer' })
  prioridade: number & Opt = 0;

  @Enum({ items: () => TransporteCarregamento })
  carregamento: TransporteCarregamento & Opt =
    TransporteCarregamento.NAO_INICIADO;

  @Enum({ items: () => TransporteConferencia })
  conferencia: TransporteConferencia & Opt = TransporteConferencia.NAO_INICIADO;

  @Enum({ items: () => TransporteSeparacao })
  separacao: TransporteSeparacao & Opt = TransporteSeparacao.NAO_INICIADO;
}

export enum TransporteStatus {
  AGUARDANDO_SEPARACAO = 'AGUARDANDO_SEPARACAO',
  EM_SEPARACAO = 'EM_SEPARACAO',
  SEPARACAO_CONCLUIDA = 'SEPARACAO_CONCLUIDA',
  EM_CONFERENCIA = 'EM_CONFERENCIA',
  CONFERENCIA_CONCLUIDA = 'CONFERENCIA_CONCLUIDA',
  EM_CARREGAMENTO = 'EM_CARREGAMENTO',
  CARREGAMENTO_CONCLUIDO = 'CARREGAMENTO_CONCLUIDO',
  FATURADO = 'FATURADO',
  LIBERADO_PORTARIA = 'LIBERADO_PORTARIA',
  CANCELADO = 'CANCELADO',
}

export enum TransporteCarregamento {
  NAO_INICIADO = 'NAO_INICIADO',
  EM_PROGRESSO = 'EM_PROGRESSO',
  CONCLUIDO = 'CONCLUIDO',
  EM_PAUSA = 'EM_PAUSA',
}

export enum TransporteConferencia {
  NAO_INICIADO = 'NAO_INICIADO',
  EM_PROGRESSO = 'EM_PROGRESSO',
  CONCLUIDO = 'CONCLUIDO',
  EM_PAUSA = 'EM_PAUSA',
}

export enum TransporteSeparacao {
  NAO_INICIADO = 'NAO_INICIADO',
  EM_PROGRESSO = 'EM_PROGRESSO',
  CONCLUIDO = 'CONCLUIDO',
  EM_PAUSA = 'EM_PAUSA',
}
