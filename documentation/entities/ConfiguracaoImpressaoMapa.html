import {
  Entity,
  Enum,
  ManyToOne,
  type Opt,
  PrimaryKey,
  Property,
  Unique,
} from '@mikro-orm/core';
import { Center } from 'src/center/domain/entities/Center.entity';
import { User } from 'src/user/domain/entities/User.entity';

@Entity({ tableName: 'ConfiguracaoImpressaoMapa' })
@Unique({
  name: 'ConfiguracaoImpressaoMapa_centerId_empresa_key',
  expression:
    'CREATE UNIQUE INDEX "ConfiguracaoImpressaoMapa_centerId_empresa_key" ON public."ConfiguracaoImpressaoMapa" USING btree ("centerId", empresa)',
  properties: ['centerId', 'empresa'],
})
export class ConfiguracaoImpressaoMapa {
  @PrimaryKey({ type: 'text' })
  id!: string;

  @Enum({
    fieldName: 'tipoImpressao',
    items: () => ConfiguracaoImpressaoMapaTipoImpressao,
  })
  tipoImpressao!: ConfiguracaoImpressaoMapaTipoImpressao;

  @Property({ fieldName: 'quebraPalete', type: 'boolean' })
  quebraPalete: boolean & Opt = false;

  @Enum({
    fieldName: 'tipoQuebra',
    items: () => ConfiguracaoImpressaoMapaTipoQuebra,
    nullable: true,
  })
  tipoQuebra?: ConfiguracaoImpressaoMapaTipoQuebra;

  @Property({
    fieldName: 'valorQuebra',
    type: 'decimal',
    precision: 65,
    scale: 30,
    nullable: true,
  })
  valorQuebra?: string;

  @Property({ fieldName: 'separarPaleteFull', type: 'boolean' })
  separarPaleteFull: boolean & Opt = false;

  @Property({ fieldName: 'separarUnidades', type: 'boolean' })
  separarUnidades: boolean & Opt = false;

  @Enum({
    fieldName: 'exibirInfoCabecalho',
    items: () => ConfiguracaoImpressaoMapaExibirInfoCabecalho,
    nullable: true,
  })
  exibirInfoCabecalho?: ConfiguracaoImpressaoMapaExibirInfoCabecalho =
    ConfiguracaoImpressaoMapaExibirInfoCabecalho.NENHUM;

  @Property({ fieldName: 'segregarFifo', nullable: true })
  segregarFifo?: string[];

  @Property({ fieldName: 'dataMaximaPercentual', type: 'integer' })
  dataMaximaPercentual: number & Opt = 0;

  @Property({
    fieldName: 'createdAt',
    type: 'datetime',
    columnType: 'timestamp(3)',
    defaultRaw: `CURRENT_TIMESTAMP`,
  })
  createdAt!: Date & Opt;

  @Property({ fieldName: 'updatedAt', columnType: 'timestamp(3)' })
  updatedAt!: Date;

  @ManyToOne({
    entity: () => Center,
    fieldName: 'centerId',
    updateRule: 'cascade',
  })
  centerId!: Center;

  @ManyToOne({
    entity: () => User,
    fieldName: 'atribuidoPorId',
    updateRule: 'cascade',
    deleteRule: 'set null',
    nullable: true,
  })
  atribuidoPorId?: User;

  @Property({ type: 'text' })
  empresa: string & Opt = 'LDB';
}

export enum ConfiguracaoImpressaoMapaTipoImpressao {
  TRANSPORTE = 'TRANSPORTE',
  CLIENTE = 'CLIENTE',
}

export enum ConfiguracaoImpressaoMapaTipoQuebra {
  LINHAS = 'LINHAS',
  PERCENTUAL = 'PERCENTUAL',
}

export enum ConfiguracaoImpressaoMapaExibirInfoCabecalho {
  PRIMEIRO = 'PRIMEIRO',
  TODOS = 'TODOS',
  NENHUM = 'NENHUM',
}
