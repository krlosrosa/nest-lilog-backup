import {
  Entity,
  Enum,
  ManyToOne,
  type Opt,
  PrimaryKeyProp,
  Property,
} from '@mikro-orm/core';
import { User } from './User.entity';
import { Center } from 'src/center/domain/entities/Center.entity';

@Entity({ tableName: 'UserCenter' })
export class UserCenter {
  [PrimaryKeyProp]?: ['userId', 'centerId', 'processo'];

  @ManyToOne({
    entity: () => User,
    fieldName: 'userId',
    updateRule: 'cascade',
    primary: true,
  })
  userId!: User;

  @ManyToOne({
    entity: () => Center,
    fieldName: 'centerId',
    updateRule: 'cascade',
    primary: true,
  })
  centerId!: Center;

  @Property({
    fieldName: 'assignedAt',
    type: 'datetime',
    columnType: 'timestamp(3)',
    defaultRaw: `CURRENT_TIMESTAMP`,
  })
  assignedAt!: Date & Opt;

  @Property({ type: 'text', primary: true }) // Mude de @PrimaryKey para @Property
  processo: string & Opt = 'EXPEDICAO';

  @Enum({ items: () => UserCenterRole })
  role: UserCenterRole & Opt = UserCenterRole.FUNCIONARIO;

  @Property({ type: 'array', nullable: true })
  roles?: string[];
}

export enum UserCenterRole {
  FUNCIONARIO = 'FUNCIONARIO',
  USER = 'USER',
  ADMIN = 'ADMIN',
  MASTER = 'MASTER',
}
